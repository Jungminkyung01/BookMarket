CREATE TABLE BOOK(
	bookID VARCHAR(10) NOT NULL,
	bookName VARCHAR(20),
	unitPrice INTEGER,
	author VARCHAR(50),
	description TEXT,
	publisher VARCHAR(20),
	category VARCHAR(20),
	unitsInStock LONG,
	totalpages LONG,
	releaseDate VARCHAR(20),
	country VARCHAR(20),
	b_condition VARCHAR(20),
	b_filename VARCHAR(20),
	PRIMARY KEY (bookID)
)default CHARSET=utf8;
